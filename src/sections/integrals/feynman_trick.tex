    \label{nahin_holzweg}%

\begin{problem}
    \label{nahin_dini}%
    Niech $a \ge 0$ będzie dowolną liczbą rzeczywistą.
    Wtedy
    \begin{equation}
        \int_0^\pi \log (1 - 2 a \cos x + a^2) \,\mathrm{d} x = \begin{cases}
            0, & \text{gdy } a^2 \le 1, \\
            2 \pi \log a & \textrm{w przeciwnym razie}.
        \end{cases}
    \end{equation}
\end{problem}
