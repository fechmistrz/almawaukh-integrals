\begin{problem}[pytanie 3490404]

    Udowodnić bez wyznaczania całek, że
    \begin{align}
        2\int_0^\frac{\pi}{2}\frac{x\log(1-\sin x)}{\sin x} \,\mathrm{d}x =
        \int_\frac{\pi}{2}^\pi\frac{x\log(1-\sin x)}{\sin x} \,\mathrm{d}x.
    \end{align}
\end{problem}
